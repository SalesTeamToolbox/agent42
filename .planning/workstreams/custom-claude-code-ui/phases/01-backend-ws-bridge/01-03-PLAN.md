    # Session management REST endpoints

    @app.get("/api/cc/sessions")
    async def cc_sessions(_user: str = Depends(get_current_user)):
        """List all CC sessions sorted by last activity (most recent first)."""
        sessions = []
        if _CC_SESSIONS_DIR.exists():
            for f in sorted(_CC_SESSIONS_DIR.glob("*.json"), key=lambda p: p.stat().st_mtime, reverse=True):
                try:
                    async with _aiofiles.open(f, "r") as fh:
                        data = _json.loads(await fh.read())
                    sessions.append(data)
                except Exception:
                    pass
        return {"sessions": sessions}

    @app.delete("/api/cc/sessions/{session_id}")
    async def cc_delete_session(session_id: str, _user: str = Depends(get_current_user)):
        """Delete a CC session file."""
        path = _CC_SESSIONS_DIR / f"{session_id}.json"
        if path.exists():
            path.unlink()
        return {"status": "ok"}

    # CC subscription detection (BRIDGE-06)

    _cc_auth_cache: dict = {"result": None, "expires": 0.0}

    async def _check_cc_subscription() -> tuple:
        """Check CC auth status with 60-second TTL cache.

        Returns (available: bool, message: str).
        Uses exit code only (stable); does not parse JSON fields (schema may change).
        """
        import time as _time
        now = _time.monotonic()
        if _cc_auth_cache["expires"] > now and _cc_auth_cache["result"] is not None:
            return _cc_auth_cache["result"]

        claude_bin = _shutil.which("claude")
        if not claude_bin:
            result = (False, "claude CLI not installed")
        else:
            try:
                proc = await _asyncio.create_subprocess_exec(
                    claude_bin, "auth", "status",
                    stdout=_asyncio.subprocess.PIPE,
                    stderr=_asyncio.subprocess.PIPE,
                )
                await _asyncio.wait_for(proc.wait(), timeout=10.0)
                result = (proc.returncode == 0, "CC subscription active" if proc.returncode == 0 else "CC not authenticated")
            except _asyncio.TimeoutError:
                result = (False, "claude auth status timed out")
            except Exception as e:
                result = (False, f"auth check failed: {e}")

        _cc_auth_cache["result"] = result
        _cc_auth_cache["expires"] = now + 60.0
        return result

    @app.get("/api/cc/auth-status")
    async def cc_auth_status(_user: str = Depends(get_current_user)):
        """Report CC subscription availability (cached 60s)."""
        available, message = await _check_cc_subscription()
        return {"available": available, "message": message}
